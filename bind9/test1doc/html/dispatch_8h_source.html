54 /*%
55  * This event is sent to a task when a response comes in.
56  * No part of this structure should ever be modified by the caller,
57  * other than parts of the buffer. The holy parts of the buffer are
58  * the base and size of the buffer. All other parts of the buffer may
59  * be used. On event delivery the used region contains the packet.
60  *
61  * "id" is the received message id,
62  *
63  * "addr" is the host that sent it to us,
64  *
65  * "buffer" holds state on the received data.
66  *
67  * The "free" routine for this event will clean up itself as well as
68  * any buffer space allocated from common pools.
69  */

94 /*%
95  * Attributes for added dispatchers.
96  *
97  * Values with the mask 0xffff0000 are application defined.
98  * Values with the mask 0x0000ffff are library defined.
99  *
100  * Insane values (like setting both TCP and UDP) are not caught. Don't
101  * do that.
102  *
103  * _PRIVATE
104  * The dispatcher cannot be shared.
105  *
106  * _TCP, _UDP
107  * The dispatcher is a TCP or UDP socket.
108  *
109  * _IPV4, _IPV6
110  * The dispatcher uses an IPv4 or IPv6 socket.
111  *
112  * _NOLISTEN
113  * The dispatcher should not listen on the socket.
114  *
115  * _MAKEQUERY
116  * The dispatcher can be used to issue queries to other servers, and
117  * accept replies from them.
118  *
119  * _RANDOMPORT
120  * Previously used to indicate that the port of a dispatch UDP must be
121  * chosen randomly. This behavior now always applies and the attribute
122  * is obsoleted.
123  *
124  * _EXCLUSIVE
125  * A separate socket will be used on-demand for each transaction.
126  */
127 #define DNS_DISPATCHATTR_PRIVATE 0x00000001U
128 #define DNS_DISPATCHATTR_TCP 0x00000002U
129 #define DNS_DISPATCHATTR_UDP 0x00000004U
130 #define DNS_DISPATCHATTR_IPV4 0x00000008U
131 #define DNS_DISPATCHATTR_IPV6 0x00000010U
132 #define DNS_DISPATCHATTR_NOLISTEN 0x00000020U
133 #define DNS_DISPATCHATTR_MAKEQUERY 0x00000040U
134 #define DNS_DISPATCHATTR_CONNECTED 0x00000080U
135 #define DNS_DISPATCHATTR_FIXEDID 0x00000100U
