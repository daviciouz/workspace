36 /*
37  * Flags affecting master file formatting. Flags 0x0000FFFF
38  * define the formatting of the rdata part and are defined in
39  * rdata.h.
40  */
41 
42 /*% Omit the owner name when possible. */
43 #define DNS_STYLEFLAG_OMIT_OWNER 0x00010000U
44 
45 /*%
46  * Omit the TTL when possible. If DNS_STYLEFLAG_TTL is
47  * also set, this means no TTLs are ever printed
48  * because $TTL directives are generated before every
49  * change in the TTL. In this case, no columns need to
50  * be reserved for the TTL. Master files generated with
51  * these options will be rejected by BIND 4.x because it
52  * does not recognize the $TTL directive.
53  *
54  * If DNS_STYLEFLAG_TTL is not also set, the TTL will be
55  * omitted when it is equal to the previous TTL.
56  * This is correct according to RFC1035, but the
57  * TTLs may be silently misinterpreted by older
58  * versions of BIND which use the SOA MINTTL as a
59  * default TTL value.
60  */
61 #define DNS_STYLEFLAG_OMIT_TTL 0x00020000U
62 
63 /*% Omit the class when possible. */
64 #define DNS_STYLEFLAG_OMIT_CLASS 0x00040000U
65 
66 /*% Output $TTL directives. */
67 #define DNS_STYLEFLAG_TTL 0x00080000U
68 
69 /*%
70  * Output $ORIGIN directives and print owner names relative to
71  * the origin when possible.
72  */
73 #define DNS_STYLEFLAG_REL_OWNER 0x00100000U
74 
75 /*% Print domain names in RR data in relative form when possible.

77 #define DNS_STYLEFLAG_REL_DATA 0x00200000U
