416 dns_message_parse(dns_message_t *msg, isc_buffer_t *source,
417  unsigned int options);
418 /*%<
419  * Parse raw wire data in 'source' as a DNS message.
420  *
421  * OPT records are detected and stored in the pseudo-section "opt".
422  * TSIGs are detected and stored in the pseudo-section "tsig".
423  *
424  * If #DNS_MESSAGEPARSE_PRESERVEORDER is set, or if the opcode of the message
425  * is UPDATE, a separate dns_name_t object will be created for each RR in the
426  * message. Each such dns_name_t will have a single rdataset containing the
427  * single RR, and the order of the RRs in the message is preserved.
428  * Otherwise, only one dns_name_t object will be created for each unique
429  * owner name in the section, and each such dns_name_t will have a list
430  * of rdatasets. To access the names and their data, use
431  * dns_message_firstname() and dns_message_nextname().
432  *
433  * If #DNS_MESSAGEPARSE_BESTEFFORT is set, errors in message content will
434  * not be considered FORMERRs. If the entire message can be parsed, it
435  * will be returned and DNS_R_RECOVERABLE will be returned.
436  *
437  * If #DNS_MESSAGEPARSE_IGNORETRUNCATION is set then return as many complete
438  * RR's as possible, DNS_R_RECOVERABLE will be returned.
439  *
440  * OPT and TSIG records are always handled specially, regardless of the
441  * 'preserve_order' setting.
442  *
443  * Requires:
444  *\li "msg" be valid.
445  *
446  *\li "buffer" be a wire format buffer.
447  *
448  * Ensures:
449  *\li The buffer's data format is correct.
450  *
451  *\li The buffer's contents verify as correct regarding header bits, buffer
452  * and rdata sizes, etc.
453  *
454  * Returns:
455  *\li #ISC_R_SUCCESS -- all is well
456  *\li #ISC_R_NOMEMORY -- no memory
457  *\li #DNS_R_RECOVERABLE -- the message parsed properly, but contained
458  * errors.
459  *\li Many other errors possible XXXMLG
460  */
