149  if (((fd_set FAR *)(set))->fd_array[__i] == (SOCKET) fd) { \

164  for (__i = 0; __i < ((fd_set FAR *)(set))->fd_count; __i++) { \

169  if (__i == ((fd_set FAR *)(set))->fd_count) { \
