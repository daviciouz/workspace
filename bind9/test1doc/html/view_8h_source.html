529  isc_boolean_t use_hints, isc_boolean_t use_static_stub,

532 /*%<
533  * Find an rdataset whose owner name is 'name', and whose type is
534  * 'type'.
535  * In general, this function first searches view's zone and cache DBs for the
536  * best match data against 'name'. If nothing found there, and if 'use_hints'
537  * is ISC_TRUE, the view's hint DB (if configured) is searched.
538  * If the view is configured with a static-stub zone which gives the longest
539  * match for 'name' among the zones, however, the cache DB is not consulted
540  * unless 'use_static_stub' is ISC_FALSE (see below about this argument).
541  *
542  * dns_view_find() is a backward compatible version equivalent to
543  * dns_view_find2() with use_static_stub argument being ISC_FALSE.
544  *
545  * Notes:
546  *
547  *\li See the description of dns_db_find() for information about 'options'.
548  * If the caller sets #DNS_DBFIND_GLUEOK, it must ensure that 'name'
549  * and 'type' are appropriate for glue retrieval.
550  *
551  *\li If 'now' is zero, then the current time will be used.
552  *
553  *\li If 'use_hints' is ISC_TRUE, and the view has a hints database, then
554  * it will be searched last. If the answer is found in the hints
555  * database, the result code will be DNS_R_HINT. If the name is found
556  * in the hints database but not the type, the result code will be
557  * #DNS_R_HINTNXRRSET.
558  *
559  *\li If 'use_static_stub' is ISC_FALSE and the longest match zone for 'name'
560  * is a static-stub zone, it's ignored and the cache and/or hints will be
561  * searched. In the majority of the cases this argument should be
562  * ISC_FALSE. The only known usage of this argument being ISC_TRUE is
563  * if this search is for a "bailiwick" glue A or AAAA RRset that may
564  * best match a static-stub zone. Consider the following example:
565  * this view is configured with a static-stub zone "example.com",
566  * and an attempt of recursive resolution needs to send a query for the
567  * zone. In this case it's quite likely that the resolver is trying to
568  * find A/AAAA RRs for the apex name "example.com". And, to honor the
569  * static-stub configuration it needs to return the glue RRs in the
570  * static-stub zone even if that exact RRs coming from the authoritative
571  * zone has been cached.
572  * In other general cases, the requested data is better to be
573  * authoritative, either locally configured or retrieved from an external
574  * server, and the data in the static-stub zone should better be ignored.
575  *
576  *\li 'foundname' must meet the requirements of dns_db_find().
577  *
578  *\li If 'sigrdataset' is not NULL, and there is a SIG rdataset which
579  * covers 'type', then 'sigrdataset' will be bound to it.
580  *
581  * Requires:
582  *
583  *\li 'view' is a valid, frozen view.
584  *
585  *\li 'name' is valid name.
586  *
587  *\li 'type' is a valid dns_rdatatype_t, and is not a meta query type
588  * except dns_rdatatype_any.
589  *
590  *\li dbp == NULL || *dbp == NULL
591  *
592  *\li nodep == NULL || *nodep == NULL. If nodep != NULL, dbp != NULL.
593  *
594  *\li 'foundname' is a valid name with a dedicated buffer or NULL.
595  *
596  *\li 'rdataset' is a valid, disassociated rdataset.
597  *
598  *\li 'sigrdataset' is NULL, or is a valid, disassociated rdataset.
599  *
600  * Ensures:
601  *
602  *\li In successful cases, 'rdataset', and possibly 'sigrdataset', are
603  * bound to the found data.
604  *
605  *\li If dbp != NULL, it points to the database containing the data.
606  *
607  *\li If nodep != NULL, it points to the database node containing the data.
608  *
609  *\li If foundname != NULL, it contains the full name of the found data.
610  *
611  * Returns:
612  *
613  *\li Any result that dns_db_find() can return, with the exception of
614  * #DNS_R_DELEGATION.
615  */

622 /*%<
623  * Find an rdataset whose owner name is 'name', and whose type is
624  * 'type'.
625  *
626  * Notes:
627  *
628  *\li This routine is appropriate for simple, exact-match queries of the
629  * view. 'name' must be a canonical name; there is no DNAME or CNAME
630  * processing.
631  *
632  *\li See the description of dns_db_find() for information about 'options'.
633  * If the caller sets DNS_DBFIND_GLUEOK, it must ensure that 'name'
634  * and 'type' are appropriate for glue retrieval.
635  *
636  *\li If 'now' is zero, then the current time will be used.
637  *
638  *\li If 'use_hints' is ISC_TRUE, and the view has a hints database, then
639  * it will be searched last. If the answer is found in the hints
640  * database, the result code will be DNS_R_HINT. If the name is found
641  * in the hints database but not the type, the result code will be
642  * DNS_R_HINTNXRRSET.
643  *
644  *\li If 'sigrdataset' is not NULL, and there is a SIG rdataset which
645  * covers 'type', then 'sigrdataset' will be bound to it.
646  *
647  * Requires:
648  *
649  *\li 'view' is a valid, frozen view.
650  *
651  *\li 'name' is valid name.
652  *
653  *\li 'type' is a valid dns_rdatatype_t, and is not a meta query type
654  * (e.g. dns_rdatatype_any), or dns_rdatatype_rrsig.
655  *
656  *\li 'rdataset' is a valid, disassociated rdataset.
657  *
658  *\li 'sigrdataset' is NULL, or is a valid, disassociated rdataset.
659  *
660  * Ensures:
661  *
662  *\li In successful cases, 'rdataset', and possibly 'sigrdataset', are
663  * bound to the found data.
664  *
665  * Returns:
666  *
667  *\li #ISC_R_SUCCESS Success; result is desired type.
668  *\li DNS_R_GLUE Success; result is glue.
669  *\li DNS_R_HINT Success; result is a hint.
