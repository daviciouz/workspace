123 /*
124  * _INET:
125  * _INET6:
126  * return addresses of that type.
127  *
128  * _EMPTYEVENT:
129  * Only schedule an event if no addresses are known.
130  * Must set _WANTEVENT for this to be meaningful.
131  *
132  * _WANTEVENT:
133  * An event is desired. Check this bit in the returned find to see
134  * if one will actually be generated.
135  *
136  * _AVOIDFETCHES:
137  * If set, fetches will not be generated unless no addresses are
138  * available in any of the address families requested.
139  *
140  * _STARTATZONE:
141  * Fetches will start using the closest zone data or use the root servers.
142  * This is useful for reestablishing glue that has expired.
143  *
144  * _GLUEOK:
145  * _HINTOK:
146  * Glue or hints are ok. These are used when matching names already
147  * in the adb, and when dns databases are searched.
148  *
149  * _RETURNLAME:
150  * Return lame servers in a find, so that all addresses are returned.
151  *
152  * _LAMEPRUNED:
153  * At least one address was omitted from the list because it was lame.
154  * This bit will NEVER be set if _RETURNLAME is set in the createfind().
155  */
156 /*% Return addresses of type INET. */
157 #define DNS_ADBFIND_INET 0x00000001
158 /*% Return addresses of type INET6. */
159 #define DNS_ADBFIND_INET6 0x00000002
160 #define DNS_ADBFIND_ADDRESSMASK 0x00000003
161 /*%
162  * Only schedule an event if no addresses are known.
163  * Must set _WANTEVENT for this to be meaningful.
164  */
165 #define DNS_ADBFIND_EMPTYEVENT 0x00000004
166 /*%
167  * An event is desired. Check this bit in the returned find to see
168  * if one will actually be generated.
169  */
170 #define DNS_ADBFIND_WANTEVENT 0x00000008
171 /*%
172  * If set, fetches will not be generated unless no addresses are
173  * available in any of the address families requested.
174  */
175 #define DNS_ADBFIND_AVOIDFETCHES 0x00000010
176 /*%
177  * Fetches will start using the closest zone data or use the root servers.
178  * This is useful for reestablishing glue that has expired.
179  */
180 #define DNS_ADBFIND_STARTATZONE 0x00000020
181 /*%
182  * Glue or hints are ok. These are used when matching names already
183  * in the adb, and when dns databases are searched.
184  */
185 #define DNS_ADBFIND_GLUEOK 0x00000040
186 /*%
187  * Glue or hints are ok. These are used when matching names already
188  * in the adb, and when dns databases are searched.

190 #define DNS_ADBFIND_HINTOK 0x00000080
191 /*%
192  * Return lame servers in a find, so that all addresses are returned.

342 /*%<
343  * Main interface for clients. The adb will look up the name given in
344  * "name" and will build up a list of found addresses, and perhaps start
345  * internal fetches to resolve names that are unknown currently.
346  *
347  * If other addresses resolve after this call completes, an event will
348  * be sent to the <task, taskaction, arg> with the sender of that event
349  * set to a pointer to the dns_adbfind_t returned by this function.
350  *
351  * If no events will be generated, the *find->result_v4 and/or result_v6
352  * members may be examined for address lookup status. The usual #ISC_R_SUCCESS,
353  * #ISC_R_FAILURE, #DNS_R_NXDOMAIN, and #DNS_R_NXRRSET are returned, along with
354  * #ISC_R_NOTFOUND meaning the ADB has not _yet_ found the values. In this
355  * latter case, retrying may produce more addresses.
356  *
357  * If events will be returned, the result_v[46] members are only valid
358  * when that event is actually returned.
359  *
360  * The list of addresses returned is unordered. The caller must impose
361  * any ordering required. The list will not contain "known bad" addresses,
362  * however. For instance, it will not return hosts that are known to be
363  * lame for the zone in question.
364  *
365  * The caller cannot (directly) modify the contents of the address list's
366  * fields other than the "link" field. All values can be read at any
367  * time, however.
368  *
369  * The "now" parameter is used only for determining which entries that
370  * have a specific time to live or expire time should be removed from
371  * the running database. If specified as zero, the current time will
372  * be retrieved and used.
373  *
374  * If 'target' is not NULL and 'name' is an alias (i.e. the name is
375  * CNAME'd or DNAME'd to another name), then 'target' will be updated with
376  * the domain name that 'name' is aliased to.
377  *
378  * All addresses returned will have the sockaddr's port set to 'port.'
379  * The caller may change them directly in the dns_adbaddrinfo_t since
380  * they are copies of the internal address only.
381  *
382  * XXXMLG Document options, especially the flags which control how
383  * events are sent.
384  *
385  * Requires:
386  *
387  *\li *adb be a valid isc_adb_t object.
388  *
389  *\li If events are to be sent, *task be a valid task,
390  * and isc_taskaction_t != NULL.
391  *
392  *\li *name is a valid dns_name_t.
393  *
394  *\li qname != NULL and *qname be a valid dns_name_t.
395  *
396  *\li target == NULL or target is a valid name with a buffer.
397  *
398  *\li find != NULL && *find == NULL.
399  *
400  * Returns:
401  *
402  *\li #ISC_R_SUCCESS Addresses might have been returned, and events will be
403  * delivered for unresolved addresses.
404  *\li #ISC_R_NOMORE Addresses might have been returned, but no events
405  * will ever be posted for this context. This is only
406  * returned if task != NULL.
407  *\li #ISC_R_NOMEMORY insufficient resources
408  *\li #DNS_R_ALIAS 'name' is an alias for another name.
409  *
410  * Calls, and returns error codes from:
411  *
412  *\li isc_stdtime_get()
413  *
414  * Notes:
415  *
416  *\li No internal reference to "name" exists after this function
417  * returns.
418  */
