430 /*%<
431  * Set high and low water marks for this memory context.
432  *
433  * When the memory usage of 'mctx' exceeds 'hiwater',
434  * '(water)(water_arg, #ISC_MEM_HIWATER)' will be called. 'water' needs to
435  * call isc_mem_waterack() with #ISC_MEM_HIWATER to acknowledge the state
436  * change. 'water' may be called multiple times.
437  *
438  * When the usage drops below 'lowater', 'water' will again be called, this
439  * time with #ISC_MEM_LOWATER. 'water' need to calls isc_mem_waterack() with
440  * #ISC_MEM_LOWATER to acknowledge the change.
441  *
442  * static void
443  * water(void *arg, int mark) {
444  * struct foo *foo = arg;
445  *
446  * LOCK(&foo->marklock);
447  * if (foo->mark != mark) {
448  * foo->mark = mark;
449  * ....
450  * isc_mem_waterack(foo->mctx, mark);
451  * }
452  * UNLOCK(&foo->marklock);
453  * }
454  *
455  * If 'water' is NULL then 'water_arg', 'hi_water' and 'lo_water' are
456  * ignored and the state is reset.
457  *
458  * Requires:
459  *
460  * 'water' is not NULL.
461  * hi_water >= lo_water
462  */
463 
464 void
465 isc_mem_waterack(isc_mem_t *ctx, int mark);

467  * Called to acknowledge changes in signaled by calls to 'water'.
