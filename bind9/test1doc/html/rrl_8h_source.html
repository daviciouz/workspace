74 /*
75  * A rate limit bucket key.
76  * This should be small to limit the total size of the database.
77  * The hash of the qname should be wide enough to make the probability
78  * of collisions among requests from a single IP address block less than 50%.
79  * We need a 32-bit hash value for 10000 qps (e.g. random qnames forged
80  * by attacker) to collide with legitimate qnames from the target with
81  * probability at most 1%.
82  */
83 #define DNS_RRL_MAX_PREFIX 64
84 typedef union dns_rrl_key dns_rrl_key_t;
85 struct dns__rrl_key {
86  isc_uint32_t ip[DNS_RRL_MAX_PREFIX/32];
87  isc_uint32_t qname_hash;
88  dns_rdatatype_t qtype;
89  isc_uint8_t qclass;
90  dns_rrl_rtype_t rtype :4; /* 3 bits + sign bit */
91  isc_boolean_t ipv6 :1;
92 };
