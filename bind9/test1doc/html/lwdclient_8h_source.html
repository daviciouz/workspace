100 /*%
101  * Client states.
102  *
103  * _IDLE The client is not doing anything at all.
104  *
105  * _RECV The client is waiting for data after issuing a socket recv().
106  *
107  * _RECVDONE Data has been received, and is being processed.
108  *
109  * _FINDWAIT An adb (or other) request was made that cannot be satisfied
110  * immediately. An event will wake the client up.
111  *
112  * _SEND All data for a response has completed, and a reply was
113  * sent via a socket send() call.
114  *
115  * Badly formatted state table:
116  *
117  * IDLE -> RECV when client has a recv() queued.
118  *
119  * RECV -> RECVDONE when recvdone event received.
120  *
121  * RECVDONE -> SEND if the data for a reply is at hand.
122  * RECVDONE -> FINDWAIT if more searching is needed, and events will
123  * eventually wake us up again.
124  *
125  * FINDWAIT -> SEND when enough data was received to reply.
126  *
127  * SEND -> IDLE when a senddone event was received.
128  *
129  * At any time -> IDLE on error. Sometimes this will be -> SEND
130  * instead, if enough data is on hand to reply with a meaningful
131  * error.
132  *
133  * Packets which are badly formatted may or may not get error returns.
134  */
135 #define NS_LWDCLIENT_STATEIDLE 1
136 #define NS_LWDCLIENT_STATERECV 2
137 #define NS_LWDCLIENT_STATERECVDONE 3
138 #define NS_LWDCLIENT_STATEFINDWAIT 4
139 #define NS_LWDCLIENT_STATESEND 5
140 #define NS_LWDCLIENT_STATESENDDONE 6
