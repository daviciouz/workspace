335 isc_result_t
336 dns_dnssec_updatekeys(dns_dnsseckeylist_t *keys, dns_dnsseckeylist_t *newkeys,
337  dns_dnsseckeylist_t *removed, dns_name_t *origin,
338  dns_ttl_t hint_ttl, dns_diff_t *diff, isc_boolean_t allzsk,
339  isc_mem_t *mctx, void (*report)(const char *, ...));
340 /*%<
341  * Update the list of keys in 'keys' with new key information in 'newkeys'.
342  *
343  * For each key in 'newkeys', see if it has a match in 'keys'.
344  * - If not, and if the metadata says the key should be published:
345  * add it to 'keys', and place a dns_difftuple into 'diff' so
346  * the key can be added to the DNSKEY set. If the metadata says it
347  * should be active, set the first_sign flag.
348  * - If so, and if the metadata says it should be removed:
349  * remove it from 'keys', and place a dns_difftuple into 'diff' so
350  * the key can be removed from the DNSKEY set. if 'removed' is non-NULL,
351  * copy the key into that list; otherwise destroy it.
352  * - Otherwise, make sure keys has current metadata.
353  *
354  * If 'allzsk' is true, we are allowing KSK-flagged keys to be used as
355  * ZSKs.
356  *
357  * 'hint_ttl' is the TTL to use for the DNSKEY RRset if there is no
358  * existing RRset, and if none of the keys to be added has a default TTL
359  * (in which case we would use the shortest one). If the TTL is longer
360  * than the time until a new key will be activated, then we have to delay
361  * the key's activation.
362  *
363  * 'report' points to a function for reporting status.
364  *
365  * On completion, any remaining keys in 'newkeys' are freed.
366  */
