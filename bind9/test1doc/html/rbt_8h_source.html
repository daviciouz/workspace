228 /*%
229  * The number of level blocks to allocate at a time. Currently the maximum
230  * number of levels is allocated directly in the structure, but future
231  * revisions of this code might have a static initial block with dynamic
232  * growth. Allocating space for 256 levels when the tree is almost never that
233  * deep is wasteful, but it's not clear that it matters, since the waste is
234  * only 2MB for 1000 concurrently active chains on a system with 64-bit
235  * pointers.
236  */
237 #define DNS_RBT_LEVELBLOCK 254

239 typedef struct dns_rbtnodechain {
240  unsigned int magic;
241  isc_mem_t * mctx;
242  /*%
243  * The terminal node of the chain. It is not in levels[].
244  * This is ostensibly private ... but in a pinch it could be
245  * used tell that the chain points nowhere without needing to
246  * call dns_rbtnodechain_current().
247  */
248  dns_rbtnode_t * end;
249  /*%
250  * The maximum number of labels in a name is 128; bitstrings mean
251  * a conceptually very large number (which I have not bothered to
252  * compute) of logical levels because splitting can potentially occur
253  * at each bit. However, DNSSEC restricts the number of "logical"
254  * labels in a name to 255, meaning only 254 pointers are needed
255  * in the worst case.
256  */
257  dns_rbtnode_t * levels[DNS_RBT_LEVELBLOCK];
258  /*%
259  * level_count indicates how deep the chain points into the
260  * tree of trees, and is the index into the levels[] array.
261  * Thus, levels[level_count - 1] is the last level node stored.
262  * A chain that points to the top level of the tree of trees has
263  * a level_count of 0, the first level has a level_count of 1, and
264  * so on.
265  */
266  unsigned int level_count;
267  /*%
268  * level_matches tells how many levels matched above the node
269  * returned by dns_rbt_findnode(). A match (partial or exact) found
270  * in the first level thus results in level_matches being set to 1.
271  * This is used by the rbtdb to set the start point for a recursive
272  * search of superdomains until the RR it is looking for is found.
273  */
274  unsigned int level_matches;
275 } dns_rbtnodechain_t;

428 /*%<
429  * Find the node for 'name'.
430  *
431  * Notes:
432  *\li A node that has no data is considered not to exist for this function,
433  * unless the DNS_RBTFIND_EMPTYDATA option is set. This applies to both
434  * exact matches and partial matches.
435  *
436  *\li If the chain parameter is non-NULL, then the path through the tree
437  * to the DNSSEC predecessor of the searched for name is maintained,
438  * unless the DNS_RBTFIND_NOPREDECESSOR or DNS_RBTFIND_NOEXACT option
439  * is used. (For more details on those options, see below.)
440  *
441  *\li If there is no predecessor, then the chain will point to nowhere, as
442  * indicated by chain->end being NULL or dns_rbtnodechain_current
443  * returning ISC_R_NOTFOUND. Note that in a normal Internet DNS RBT
444  * there will always be a predecessor for all names except the root
445  * name, because '.' will exist and '.' is the predecessor of
446  * everything. But you can certainly construct a trivial tree and a
447  * search for it that has no predecessor.
448  *
449  *\li Within the chain structure, the 'levels' member of the structure holds
450  * the root node of each level except the first.
451  *
452  *\li The 'level_count' of the chain indicates how deep the chain to the
453  * predecessor name is, as an index into the 'levels[]' array. It does
454  * not count name elements, per se, but only levels of the tree of trees,
455  * the distinction arising because multiple labels from a name can be
456  * stored on only one level. It is also does not include the level
457  * that has the node, since that level is not stored in levels[].
458  *
459  *\li The chain's 'level_matches' is not directly related to the predecessor.
460  * It is the number of levels above the level of the found 'node',
461  * regardless of whether it was a partial match or exact match. When
462  * the node is found in the top level tree, or no node is found at all,
463  * level_matches is 0.
464  *
465  *\li When DNS_RBTFIND_NOEXACT is set, the closest matching superdomain is
466  * returned (also subject to DNS_RBTFIND_EMPTYDATA), even when
467  * there is an exact match in the tree. In this case, the chain
468  * will not point to the DNSSEC predecessor, but will instead point
469  * to the exact match, if there was any. Thus the preceding paragraphs
470  * should have "exact match" substituted for "predecessor" to describe
471  * how the various elements of the chain are set. This was done to
472  * ensure that the chain's state was sane, and to prevent problems that
473  * occurred when running the predecessor location code under conditions
474  * it was not designed for. It is not clear *where* the chain should
475  * point when DNS_RBTFIND_NOEXACT is set, so if you end up using a chain
476  * with this option because you want a particular node, let us know
477  * where you want the chain pointed, so this can be made more firm.
478  *
479  * Requires:
480  *\li rbt is a valid rbt manager.
481  *\li dns_name_isabsolute(name) == TRUE.
482  *\li node != NULL && *node == NULL.
483  *\li #DNS_RBTFIND_NOEXACT and DNS_RBTFIND_NOPREDECESSOR are mutually
484  * exclusive.
485  *
486  * Ensures:
487  *\li 'name' and the tree are not altered in any way.
488  *
489  *\li If result is ISC_R_SUCCESS:
490  *\verbatim
491  * *node is the terminal node for 'name'.
492 
493  * 'foundname' and 'name' represent the same name (though not
494  * the same memory).
495 
496  * 'chain' points to the DNSSEC predecessor, if any, of 'name'.
497  *
498  * chain->level_matches and chain->level_count are equal.
499  *\endverbatim
500  *
501  * If result is DNS_R_PARTIALMATCH:
502  *\verbatim
503  * *node is the data associated with the deepest superdomain
504  * of 'name' which has data.
505  *
506  * 'foundname' is the name of deepest superdomain (which has
507  * data, unless the DNS_RBTFIND_EMPTYDATA option is set).
508  *
509  * 'chain' points to the DNSSEC predecessor, if any, of 'name'.
510  *\endverbatim
511  *
512  *\li If result is ISC_R_NOTFOUND:
513  *\verbatim
514  * Neither the name nor a superdomain was found. *node is NULL.
515  *
516  * 'chain' points to the DNSSEC predecessor, if any, of 'name'.
517  *
518  * chain->level_matches is 0.
519  *\endverbatim
520  *
521  * Returns:
522  *\li #ISC_R_SUCCESS Success
523  *\li #DNS_R_PARTIALMATCH Superdomain found with data
524  *\li #ISC_R_NOTFOUND No match, or superdomain with no data
525  *\li #ISC_R_NOSPACE Concatenating nodes to form foundname failed
526  */
