44 /*
45  * Create a dns64 record which is used to identify the set of clients
46  * it applies to and how to perform the DNS64 synthesis.
47  *
48  * 'prefix' and 'prefixlen' defined the leading bits of the AAAA records
49  * to be synthesised. 'suffix' defines the bits after the A records bits.
50  * If suffix is NULL zeros will be used for these bits. 'client' defines
51  * for which clients this record applies. If 'client' is NULL then all
52  * clients apply. 'mapped' defines which A records are candidated for
53  * mapping. If 'mapped' is NULL then all A records will be mapped.
54  * 'excluded' defines which AAAA are to be treated as non-existent for the
55  * purposed of determining whether to perform syntesis. If 'excluded' is
56  * NULL then no AAAA records prevent synthesis.
57  *
58  * If DNS_DNS64_RECURSIVE_ONLY is set then the record will only match if
59  * DNS_DNS64_RECURSIVE is set when calling dns_dns64_aaaaok() and
60  * dns_dns64_aaaafroma().
61  *
62  * If DNS_DNS64_BREAK_DNSSEC is set then the record will still apply if
63  * DNS_DNS64_DNSSEC is set when calling dns_dns64_aaaaok() and
64  * dns_dns64_aaaafroma() otherwise the record will be ignored.
65  *
66  * Requires:
67  * 'mctx' to be valid.
68  * 'prefix' to be valid and the address family to AF_INET6.
69  * 'prefixlen' to be one of 32, 40, 48, 56, 72 and 96.
70  * the bits not covered by prefixlen in prefix to
71  * be zero.
72  * 'suffix' to be NULL or the address family be set to AF_INET6
73  * and the leading 'prefixlen' + 32 bits of the 'suffix'
74  * to be zero. If 'prefixlen' is 40, 48 or 56 then the
75  * the leading 'prefixlen' + 40 bits of 'suffix' must be
76  * zero.
77  * 'client' to be NULL or a valid acl.
78  * 'mapped' to be NULL or a valid acl.
79  * 'excluded' to be NULL or a valid acl.
80  *
81  * Returns:
82  * ISC_R_SUCCESS
83  * ISC_R_NOMEMORY
84  */

98 /*
99  * dns_dns64_aaaafroma() determines whether to perform a DNS64 address
100  * synthesis from 'a' based on 'dns64', 'reqaddr', 'reqsigner', 'env',
101  * 'flags' and 'aaaa'. If synthesis is performed then the result is
102  * written to '*aaaa'.
103  *
104  * The synthesised address will be of the form:
105  *
106  * <prefix bits><a bits><suffix bits>
107  *
108  * If <a bits> straddle bits 64-71 of the AAAA record, then 8 zero bits will
109  * be inserted at bits 64-71.
110  *
111  * Requires:
112  * 'dns64' to be valid.
113  * 'reqaddr' to be valid.
114  * 'reqsigner' to be NULL or valid.
115  * 'env' to be valid.
116  * 'a' to point to a IPv4 address in network order.
117  * 'aaaa' to point to a IPv6 address buffer in network order.
118  *
119  * Returns:
120  * ISC_R_SUCCESS if synthesis was performed.
121  * DNS_R_DISALLOWED if there is no match.
122  */
